### 1. 数据集ID自动生成

**优化前**：
- 用户需要手动输入数据集编码
- 容易出现命名不规范、重复等问题

**优化后**：
- ✅ 系统自动生成数据集编码
- ✅ 生成规则：`ds_{数据源名}_{表名}`
- ✅ 字段设置为只读，防止手动修改
- ✅ 在进入步骤3时自动生成

**生成逻辑**：
```javascript
// 格式: ds_{数据源名}_{表名}
// 示例: 数据源"MaxCompute-生产", 表"dwd_order_info"
// 生成: ds_maxcomputeprod_dwd_order_info

function autoGenerateDatasetCode() {
    const datasourceId = document.getElementById('datasourceSelect').value;
    const datasourceName = document.getElementById('datasourceSelect').selectedOptions[0]?.text || 'ds';
    const database = document.getElementById('databaseSelect').value;
    const table = document.getElementById('tableSelect').value;
    
    if (datasourceId && table) {
        // 去除特殊字符，转为小写，空格替换为下划线
        const cleanDsName = datasourceName.replace(/[^a-zA-Z0-9_]/g, '').toLowerCase() || `ds${datasourceId}`;
        const cleanTableName = table.split('.').pop().replace(/[^a-zA-Z0-9_]/g, '').toLowerCase();
        const code = `ds_${cleanDsName}_${cleanTableName}`;
        
        document.getElementById('datasetCode').value = code;
    }
}
```

## 🔄 飞书SSO集成规划

### 接入步骤

#### 1. 后端配置飞书应用
```python
# app/config.py
FEISHU_APP_ID = os.getenv('FEISHU_APP_ID')
FEISHU_APP_SECRET = os.getenv('FEISHU_APP_SECRET')
FEISHU_SSO_ENABLED = os.getenv('FEISHU_SSO_ENABLED', 'false').lower() == 'true'
```

#### 2. 创建用户会话管理
```python
# app/services/auth.py
from flask import session

class AuthService:
    @staticmethod
    def get_current_user():
        """获取当前登录用户"""
        if 'feishu_user' in session:
            return session['feishu_user']
        return {'name': 'admin', 'user_id': 'admin'}
    
    @staticmethod
    def set_current_user(user_info):
        """设置当前登录用户"""
        session['feishu_user'] = user_info
```

#### 3. 添加飞书登录路由
```python
# app/routes/auth.py
from flask import Blueprint, redirect, request, session
from ..services.feishu import FeishuClient

bp = Blueprint('auth', __name__)

@bp.route('/auth/feishu/login')
def feishu_login():
    """飞书登录重定向"""
    client = FeishuClient()
    auth_url = client.get_auth_url()
    return redirect(auth_url)

@bp.route('/auth/feishu/callback')
def feishu_callback():
    """飞书登录回调"""
    code = request.args.get('code')
    client = FeishuClient()
    user_info = client.get_user_info(code)
    
    # 存储用户信息到session
    AuthService.set_current_user(user_info)
    
    return redirect('/')
```

#### 4. 前端获取当前用户
```javascript
// 修改 getCurrentUser 函数
async function getCurrentUser() {
    try {
        const response = await fetch('/api/auth/current_user');
        const data = await response.json();
        
        if (data.code === 0 && data.data) {
            return data.data.name || data.data.user_id;
        }
    } catch (error) {
        console.error('获取当前用户失败:', error);
    }
    
    // 降级方案
    return 'admin';
}
```

#### 5. 添加用户信息API
```python
# app/routes/auth.py
@bp.route('/api/auth/current_user')
def current_user():
    """获取当前用户信息"""
    user = AuthService.get_current_user()
    return jsonify({
        'code': 0,
        'data': user
    })
```
